import java.util.ArrayDeque;
import java.util.Queue;
import java.util.Scanner;

class Queue_stack {
	int MaxSize;
	Queue<Integer> queue = new ArrayDeque<>();

	void create_stack(int size) {
		MaxSize = size;
	}

	void push(int e) {
		queue.add(e);
		System.out.println(e + " pushed successfully");
	}

	int pop() {
		if (queue.isEmpty()) {
			System.out.println("Stack is empty!");
			return -1;
		}

		for (int i = 0; i < queue.size() - 1; i++) {
			queue.add(queue.remove());
		}
		int temp = queue.peek();
		queue.remove();
		return temp;
	}

	int peek() {
		if (queue.isEmpty()) {
			System.out.println("Stack is empty!");
			return -1;
		}

		for (int i = 0; i < queue.size() - 1; i++) {
			queue.add(queue.remove());
		}
		int temp = queue.peek();
		queue.add(queue.remove()); // restore
		return temp;
	}

	boolean is_empty() {
		return queue.isEmpty();
	}

	boolean is_full() {
		return MaxSize == queue.size();
	}

	void print_stack() {
		if (queue.isEmpty()) {
			System.out.println("Stack is empty!");
			return;
		}

		Queue<Integer> temp = new ArrayDeque<>(queue);
		System.out.print("Stack contents (top â†’ bottom): ");
		for (int val : temp) {
			System.out.print(val + " ");
		}
		System.out.println();
	}
}

import java.util.ArrayDeque;
import java.util.Deque;
import java.util.Queue;
import java.util.Scanner;

class load {

	Deque<Integer> stack = new ArrayDeque<>();
	Deque<Integer> auxstack = new ArrayDeque<>();

	void enqueue(int x) {
		stack.push(x);
		System.out.println(x + " enqueued successfully");
	}

	int dequeue() {

		if (stack.isEmpty()) {
			System.out.println("Queue is empty!");
			return -1;
		}

		while (!stack.isEmpty()) {
			auxstack.push(stack.pop());
		}

		int a = auxstack.pop();

		while (!auxstack.isEmpty()) {
			stack.push(auxstack.pop());

		}
		return a;
	}



}
